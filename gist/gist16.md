            var uc = ((hs - 0xd800) * 0x400) + (ls - 0xdc00) + 0x10000;

function recursion($arr,&$res=[]){
	foreach ($arr as $key => $value) {
		if (is_array($value)) {
			recursion($value,$res);
		}else{
			$res[]=$value;
		}
	}
